class Main {
    public static void main(String[] args) {
        int nums[] = {9,1,5,3,2,7};
        int n = nums.length;
    for(int i=0; i<n; i++){
        for(int j = i+1; j<n;j++){
            if(nums[i]>nums[j]){
                int temp = nums[i];
                nums[i] = nums[j];
                nums[j] = temp;
            }
        }
    }
    System.out.println(nums[n-2]);
        
    }
}
